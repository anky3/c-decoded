Sum, Difference, Product & Division of a and b 

			#include<stdio.h>
			#include<conio.h>
			void main()
			{
				int a,b,c;
				clrscr();
				printf("Enter value of a: ");
				scanf("%d", &a);
				printf("Enter value of b: ");
				scanf("%d", &b);
				c=a+b;
				printf("\nSum = %d", a, b, c);
				c=a-b;
				printf("\nDifference = %d", a, b, c);
				c=a*b;
				printf("\nProduct = %d", a, b, c);
				c=a/b;
				printf("\nQuotent = %d", a, b, c);
				getch();
			}
