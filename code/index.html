Sum, Difference, Product & Division of a and b 

			#include<stdio.h>
			#include<conio.h>
			void main()
			{
				int a,b,c;
				clrscr();
				printf("Enter value of a: ");
				scanf("%d", &a);
				printf("\nEnter value of b: ");
				scanf("%d", &b);
				c=a+b;
				printf("\nSum = %d", c);
				c=a-b;
				printf("\nDifference = %d", c);
				c=a*b;
				printf("\nProduct = %d", c);
				c=a/b;
				printf("\nQuotent = %d", c);
				getch();
			}
